#DB LINK
자료가 있는 오라클과 내 오라클
DB LINK로 다른 DB의 데이터를 땡겨올 수 있다.

자료가 있는 대상에
CREATE DATABASE LINK KOPO_SERVER
CONNECT TO kopo
IDENTIFIED BY kopo
USING 'KOPO';

내거에
CREATE DATABASE LINK KOPO_SERVER2
CONNECT TO kopo
IDENTIFIED BY kopo
USING '(DESCRIPTION =
    (ADDRESS = (PROTOCOL = TCP)(HOST = 192.168.110.112)(PORT = 1521))
    (CONNECT_DATA =
      (SERVER = DEDICATED)
      (SERVICE_NAME = orcl)
    )
  )';

링크 확인
SELECT * FROM ALL_DB_LINKS;

링크 내용 조회법
SELECT * FROM KOPO_STUDENT_2018@KOPO_SERVER2;

SELECT * FROM 원본의 테이블명@서버명;

데이터베이스 링크 삭제방법
DROP DATABASE LINK ASUS_LINK;

DROP DATABASE LINK서버명;
